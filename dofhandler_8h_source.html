438 class DynamicFEDofHandler : public DofHandler {
439  public:
463  template <typename LOCALDOFINFO>
464  DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
465  LOCALDOFINFO &&locdof)
466  : mesh_p_(std::move(mesh_p)) {
467  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
468 
469  // Index counter for global shape functions = global dof
470  gdof_idx_t dof_idx = 0;
471 
472  // Step I: Set indices for shape functions on nodes
473  // Run through node indices (entities of co-dimension 2)
474  const size_type no_nodes = mesh_p_->NumEntities(2);
475  num_int_dofs_[2].resize(no_nodes, 0);
476  offsets_[2].resize(no_nodes + 1, 0);
477  // Traverse nodes (co-dimension-2 entities) based on indices
478  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
479  // Obtain pointer to node entity
480  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
481  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
482  // Offset for indices of node in index vector
483  glb_idx_t node_dof_offset = dof_idx;
484  offsets_[2][node_idx] = node_dof_offset;
485  // Request number of local shape functions associated with the node
486  size_type no_int_dof_node = locdof(*node_p);
487  num_int_dofs_[2][node_idx] = no_int_dof_node;
488 
489  // Store dof indices in array
490  for (unsigned j = 0; j < no_int_dof_node; j++) {
491  dofs_[2].push_back(dof_idx);
492  dof_entities_.push_back(node_p); // Store entity for current dof
493  dof_idx++; // Move on to next index
494  }
495  }
496  // Set sentinel
497  offsets_[2][no_nodes] = dof_idx;
498 
499  // Step II: Set indices for shape functions on edges (co-dimension = 1)
500  const size_type no_edges = mesh_p_->NumEntities(1);
501  // Set length of edge-related index vectors
502  num_int_dofs_[1].resize(no_edges, 0);
503  offsets_[1].resize(no_edges + 1, 0);
504  // points to the position of the current dof index
505  size_type edge_dof_offset = 0;
506  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
507  // Obtain pointer to edge entity
508  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
509  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
510  // Offset for indices of edge dof in index vector
511  offsets_[1][edge_idx] = edge_dof_offset;
512  size_type no_int_dof_edge = locdof(*edge);
513  num_int_dofs_[1][edge_idx] = no_int_dof_edge;
514 
515  // Obtain indices for basis functions sitting at endpoints
516  // Endpoints are mesh entities with co-dimension = 2
517  for (const lf::mesh::Entity *endpoint : edge->SubEntities(1)) {
518  const glb_idx_t ep_idx(mesh_p_->Index(*endpoint));
519  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
520  size_type no_int_dofs_ep = num_int_dofs_[2][ep_idx];
521  // Copy indices of shape functions from nodes to edge
522  for (unsigned j = 0; j < no_int_dofs_ep; j++) {
523  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
524  edge_dof_offset++;
525  }
526  }
527  // Set indices for interior edge degrees of freedom
528  for (unsigned j = 0; j < no_int_dof_edge; j++) {
529  dofs_[1].push_back(dof_idx);
530  edge_dof_offset++;
531  dof_entities_.push_back(edge);
532  dof_idx++;
533  }
534  }
535  // Set sentinel
536  offsets_[1][no_edges] = edge_dof_offset;
537 
538  // Step III: Set indices for shape functions on cells (co-dimension = 0)
539  const size_type no_cells = mesh_p_->NumEntities(0);
540  // Set length of cell-related index vectors
541  num_int_dofs_[0].resize(no_cells, 0);
542  offsets_[0].resize(no_cells + 1, 0);
543  // points to the position of the current dof index
544  size_type cell_dof_offset = 0;
545  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
546  // Obtain pointer to current ell
547  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
548  // Offset for indices of cell dof in index vector
549  offsets_[0][cell_idx] = cell_dof_offset;
550  size_type no_int_dof_cell = locdof(*cell);
551  num_int_dofs_[0][cell_idx] = no_int_dof_cell;
552 
553  // Obtain indices for basis functions in vertices
554  for (const lf::mesh::Entity *vertex : cell->SubEntities(2)) {
555  const glb_idx_t vt_idx(mesh_p_->Index(*vertex));
556  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
557  size_type no_int_dofs_vt = num_int_dofs_[2][vt_idx];
558  // Copy indices of shape functions from nodes to cell
559  for (unsigned j = 0; j < no_int_dofs_vt; j++) {
560  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
561  cell_dof_offset++;
562  }
563  }
564 
565  // Collect indices of interior shape functions of edges
566  // Internal ordering will depend on the orientation of the edge
567  auto edge_orientations = cell->RelativeOrientations();
568  auto edges = cell->SubEntities(1);
569  // Loop over edges
570  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
571  ed_sub_idx++) {
572  const glb_idx_t edge_idx = mesh_p_->Index(*edges[ed_sub_idx]);
573  const size_type no_int_dof_edge = num_int_dofs_[1][edge_idx];
574  const glb_idx_t edge_int_dof_offset =
575  offsets_[1][edge_idx + 1] - no_int_dof_edge;
576 
577  // Copy indices of shape functions from edges to cell
578  // The order, in which they are copied depends on the relative
579  // orientation of the edge w.r.t. the cell
580  switch (edge_orientations[ed_sub_idx]) {
581  case lf::mesh::Orientation::positive: {
582  for (int j = 0; j < no_int_dof_edge; j++) {
583  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
584  cell_dof_offset++;
585  }
586  break;
587  }
588  case lf::mesh::Orientation::negative: {
589  for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
590  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
591  cell_dof_offset++;
592  }
593  break;
594  }
595  } // end switch
596  } // end loop over edges
597 
598  // Set indices for interior shape functions of the cell
599  for (unsigned j = 0; j < no_int_dof_cell; j++) {
600  dofs_[0].push_back(dof_idx);
601  cell_dof_offset++;
602  dof_entities_.push_back(cell);
603  dof_idx++;
604  } // end loop over interior dof of cell
605  } // end loop over cells
606  // Set sentinel
607  offsets_[0][no_cells] = cell_dof_offset;
608 
609  // Finally set number of global shape functions
610  num_dof_ = dof_idx;
611  } // end constructor
