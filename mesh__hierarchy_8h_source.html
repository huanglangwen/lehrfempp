472  (edge_marked_.back())[edge_index] = marker(finest_mesh, *edge);

506  EntityCenterPositionSelector(const EntityCenterPositionSelector &) = default;
507  EntityCenterPositionSelector(EntityCenterPositionSelector &&) noexcept =

518  explicit EntityCenterPositionSelector(POSPRED pos_pred)
519  : pos_pred_(std::move(pos_pred)) {}
524  bool operator()(const lf::mesh::Entity &ent) const {
525  const lf::base::RefEl ref_el_type = ent.RefEl();
526  // Obtain shape of entity
527  lf::geometry::Geometry *geo_ptr = ent.Geometry();
528  LF_ASSERT_MSG(geo_ptr != nullptr, "Missing geometry for " << ent);
529  switch (ref_el_type) {
530  case lf::base::RefEl::kPoint(): {
531  const Eigen::MatrixXd pos(geo_ptr->Global(kpoint_center_));
532  return pos_pred_(pos);
533  }
534  case lf::base::RefEl::kSegment(): {
535  const Eigen::MatrixXd pos(geo_ptr->Global(kedge_center_));
536  return pos_pred_(pos);
537  }
538  case lf::base::RefEl::kTria(): {
539  const Eigen::MatrixXd pos(geo_ptr->Global(ktria_center_));
540  return pos_pred_(pos);
541  }
542  case lf::base::RefEl::kQuad(): {
543  const Eigen::MatrixXd pos(geo_ptr->Global(kquad_center_));
544  return pos_pred_(pos);
545  }
546  default: {
547  LF_ASSERT_MSG(false, "Illegal entity type");
548  break;
549  }
550  } // end switch
551  return false;
552  }
