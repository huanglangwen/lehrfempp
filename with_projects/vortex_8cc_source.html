35  const std::shared_ptr<const lf::mesh::Mesh> &mesh,
36  const lf::assemble::DofHandler &dofh, bool modified = false) {
37  // No volume forces are present in this experiment
38  auto f = [](const Eigen::Vector2d & /*unused*/) -> Eigen::Vector2d {
39  return Eigen::Vector2d::Zero();
40  };
41  // The top lid is driven with velocity 1
42  auto dirichlet_funct = [](const lf::mesh::Entity &edge) -> Eigen::Vector2d {
43  static constexpr double eps = 1e-10;
44  const auto geom = edge.Geometry();
45  const auto vertices = geom->Global(edge.RefEl().NodeCoords());
46  Eigen::Vector2d v;
47  v << 1. / 100, 0;
48  if (vertices(1, 0) <= 100 + eps && vertices(1, 0) >= 100 - eps &&
49  vertices(1, 1) <= 100 + eps && vertices(1, 1) >= 100 - eps) {
50  return v;
51  }
52  return Eigen::Vector2d::Zero();
53  };
54 
55  // Solve the LSE using sparse cholesky
56  const auto [A, rhs] =
57  projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
58  mesh, dofh, f, dirichlet_funct, 1,
59  lf::quad::make_TriaQR_MidpointRule(), modified);
60  Eigen::SparseMatrix<double> As = A.makeSparse();
61  Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
62  solver.compute(As);
63  return solver.solve(rhs);
