23 std::shared_ptr<mesh::Mesh> TorusMeshBuilder::Build() {
24  using coord_t = Eigen::Vector3d;
25 
26  const size_type nx = num_of_x_cells_;
27  const size_type ny = num_of_y_cells_;
28 
29  // opposite edges and vertices are identical and only counted once
30  const unsigned no_of_cells = nx * ny;
31  const unsigned no_of_edges = 2 * nx * ny;
32  const unsigned no_of_vertices = nx * ny;
33 
34  if (no_of_cells == 0) {
35  return nullptr;
36  }
37 
38  const double x_size = top_right_corner_[0] - bottom_left_corner_[0];
39  const double y_size = top_right_corner_[1] - bottom_left_corner_[1];
40 
41  if ((x_size <= 0.0) || (y_size <= 0.0)) {
42  return nullptr;
43  }
44 
45  // calculate mesh width of uniform grid on rectangle
46  const double hx = x_size / nx;
47  const double hy = y_size / ny;
48 
49  // parametrize torus: https://en.wikipedia.org/wiki/Torus#Geometry
50  const double r = x_size / (2. * M_PI);
51  const double R = y_size / (2. * M_PI);
52  auto theta = [r, hx](double i) -> double { return (i * hx) / r; };
53  auto phi = [R, hy](double j) -> double { return (j * hy) / R; };
54 
55  if (output_ctrl_ > 0) {
56  std::cout << "TorusMesh: " << no_of_cells << " cells, " << no_of_edges
57  << " edges " << no_of_vertices << " vertices, "
58  << "mesh widths hx/hy = " << hx << "/" << hy << std::endl;
59  }
60 
61  // compute vertices of mesh on torus with lexicographic numbering
62  std::vector<size_type> v_idx(no_of_vertices);
63  int node_cnt = 0;
64 
65  for (size_type j = 0; j < ny; ++j) {
66  for (size_type i = 0; i < nx; ++i, ++node_cnt) {
67  // compute vertex coordinates
68  coord_t node_coord;
69  node_coord << (R + r * std::cos(theta(i))) * std::cos(phi(j)),
70  (R + r * std::cos(theta(i))) * std::sin(phi(j)),
71  r * std::sin(theta(i));
72 
73  if (output_ctrl_ > 0) {
74  std::cout << "Adding vertex " << node_cnt << ": "
75  << node_coord.transpose() << std::endl;
76  }
77  // register vertex
78  v_idx[node_cnt] = mesh_factory_->AddPoint(node_coord);
79  }
80  }
81 
82  // compute cells of mesh on torus
83  std::vector<size_type> t_idx(no_of_cells);
84  size_type quad_cnt = 0;
85 
86  for (size_type i = 0; i < nx; ++i) {
87  for (size_type j = 0; j < ny; ++j, quad_cnt++) {
88  // gather vertex indices of given cell wrapping around edges of rectangle
89  std::vector<size_type> vertex_index_list{
90  v_idx[VertexIndex(i, j)], v_idx[VertexIndex((i + 1) % nx, j)],
91  v_idx[VertexIndex((i + 1) % nx, (j + 1) % ny)],
92  v_idx[VertexIndex(i, (j + 1) % ny)]};
93 
94  Eigen::Matrix<double, 3, 4> quad_geo;
95  quad_geo << (R + r * std::cos(theta(i))) * std::cos(phi(j)),
96  (R + r * std::cos(theta(i + 1))) * std::cos(phi(j)),
97  (R + r * std::cos(theta(i + 1))) * std::cos(phi(j + 1)),
98  (R + r * std::cos(theta(i))) * std::cos(phi(j + 1)),
99  (R + r * std::cos(theta(i))) * std::sin(phi(j)),
100  (R + r * std::cos(theta(i + 1))) * std::sin(phi(j)),
101  (R + r * std::cos(theta(i + 1))) * std::sin(phi(j + 1)),
102  (R + r * std::cos(theta(i))) * std::sin(phi(j + 1)),
103  r * std::sin(theta(i)), r * std::sin(theta(i + 1)),
104  r * std::sin(theta(i + 1)), r * std::sin(theta(i));
105 
106  if (output_ctrl_ > 0) {
107  std::cout << "Adding quad " << quad_cnt << ": " << quad_geo
108  << std::endl;
109  }
110 
111  // request cell production from MeshFactory (straight edges will be
112  // created as needed)
113  t_idx[quad_cnt] = mesh_factory_->AddEntity(
114  lf::base::RefEl::kQuad(), vertex_index_list,
115  std::make_unique<geometry::QuadO1>(quad_geo));
116  }
117  }
118 
119  return mesh_factory_->Build();
120 } // TorusMeshBuilder::Build()
