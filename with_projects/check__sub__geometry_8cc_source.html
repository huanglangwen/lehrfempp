17  // nodeCoords is a (refEl.Dimension, refEl.NumNodes) matrix
18  const auto refEl = geom.RefEl();
19  const Eigen::MatrixXd &nodeCoords = refEl.NodeCoords();
20 
21  // iterate over all relative codimensions
22  for (size_t codim = 0; codim <= geom.RefEl().Dimension(); ++codim) {
23  // iterate over all subEntities in given codimension
24  const auto numSubEntities = refEl.NumSubEntities(codim);
25  for (size_t subEntity = 0; subEntity < numSubEntities; ++subEntity) {
26  // subNodeCoords is a (subRefEl.Dimension, subRefEl.NumNodes) matrix
27  auto subGeom = geom.SubGeometry(codim, subEntity);
28  auto subRefEl = subGeom->RefEl();
29  const Eigen::MatrixXd &subNodeCoords = subRefEl.NodeCoords();
30 
31  // iterate over all nodes of subEntity
32  for (size_t subNode = 0; subNode < subRefEl.NumNodes(); ++subNode) {
33  // map coordinates in subRefEl.Dimension to geom.DimGlobal
34  auto globalCoordsFromSub = subGeom->Global(subNodeCoords.col(subNode));
35  // get index of subSubEntity with respect to refEl
36  int subSubIdx = refEl.SubSubEntity2SubEntity(
37  codim, subEntity, geom.DimLocal() - codim, subNode);
38  // map coordinates in RefEl.Dimension to geom.DimGlobal
39  auto globalCoords = geom.Global(nodeCoords.col(subSubIdx));
40 
41  EXPECT_TRUE(globalCoordsFromSub.isApprox(globalCoords))
42  << "Global mapping of subNode " << subNode << " of subEntity "
43  << subEntity << " in relative codim " << codim
44  << " differs from global mapping of node " << subSubIdx;
45  }
46 
47  // check points inside subEntity
48  if (subRefEl != lf::base::RefEl::kPoint()) {
49  // select nodes of RefEl referenced by subEntity
50  Eigen::MatrixXd mapNodeCoords(nodeCoords.rows(), subRefEl.NumNodes());
51  for (size_t subNode = 0; subNode < subRefEl.NumNodes(); ++subNode) {
52  const auto subSubIdx = refEl.SubSubEntity2SubEntity(
53  codim, subEntity, geom.DimLocal() - codim, subNode);
54  mapNodeCoords.col(subNode) = nodeCoords.col(subSubIdx);
55  }
56 
57  auto points = qrProvider(subRefEl).Points();
58 
59  // compute mapping: alpha * subNodeCoords + beta = mapNodeCoords
60  Eigen::MatrixXd paddedSubNodeCoords = Eigen::MatrixXd::Ones(
61  subNodeCoords.rows() + 1, subNodeCoords.cols());
62  paddedSubNodeCoords.block(0, 0, subNodeCoords.rows(),
63  subNodeCoords.cols()) = subNodeCoords;
64  Eigen::MatrixXd alphaBeta = paddedSubNodeCoords.transpose()
65  .fullPivLu()
66  .solve(mapNodeCoords.transpose())
67  .transpose();
68 
69  // map points onto RefEl
70  Eigen::MatrixXd paddedPoints =
71  Eigen::MatrixXd::Ones(points.rows() + 1, points.cols());
72  paddedPoints.block(0, 0, points.rows(), points.cols()) = points;
73  const Eigen::MatrixXd mappedPoints = alphaBeta * paddedPoints;
74 
75  // map coordinates in subRefEl.Dimension to geom.DimGlobal
76  auto globalPointsFromSub = subGeom->Global(points);
77  // map coordinates in RefEl.Dimension to geom.DimGlobal
78  auto globalPoints = geom.Global(mappedPoints);
79 
80  EXPECT_TRUE(globalPoints.isApprox(globalPointsFromSub))
81  << "Global mapping of points " << points << " from subEntity "
82  << subEntity << " in relative codim " << codim
83  << " differs from global mapping";
84  }
