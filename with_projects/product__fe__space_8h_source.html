207 // Initialization method.
208 template <typename SCALAR>
209 void ProductUniformFESpace<SCALAR>::init() {
210  // check validity and consistency of mesh pointer:
211  LF_ASSERT_MSG(mesh_p_ != nullptr, "No mesh");
212  LF_ASSERT_MSG(mesh_p_->DimMesh() == 2, "Only 2D cells");
213 
214  // we do not check, wheter a specification of reference finite elements
215  // for cells (triangles and quadrilaterals) is given, since this may not
216  // be desired necessary for some components in a dpg method (fluxes)
217 
218  // check validity and consistency of the number of components:
219  LF_ASSERT_MSG(numComponents_ != 0, "0 components FE space.");
220  LF_ASSERT_MSG(numComponents_ == rfs_tria_v_.size() &&
221  numComponents_ == rfs_quad_v_.size() &&
222  numComponents_ == rfs_edge_v_.size() &&
223  numComponents_ == rfs_point_v_.size(),
224  "Missmatch in number of shape functions passed for different "
225  "entity tpyes.");
226 
227  // resize vectors saving the number of loal shape functions for each
228  // component:
229  num_rsf_tria_.resize(numComponents_);
230  num_rsf_quad_.resize(numComponents_);
231  num_rsf_edge_.resize(numComponents_);
232  num_rsf_node_.resize(numComponents_);
233 
234  // compatibility checks and initialization of number of shape functions:
235  for (size_type component = 0; component < numComponents_; component++) {
236  num_rsf_tria_[component] = 0;
237  num_rsf_quad_[component] = 0;
238  num_rsf_edge_[component] = 0;
239  num_rsf_node_[component] = 0;
240 
241  if (rfs_tria_v_[component] != nullptr) {
242  // probe reference finite element
243  LF_ASSERT_MSG(rfs_tria_v_[component]->RefEl() == lf::base::RefEl::kTria(),
244  "wrong type for triangle in component" << component);
245  // initialize number of shape functions associated to entitites.
246  num_rsf_node_[component] =
247  rfs_tria_v_[component]->NumRefShapeFunctions(2);
248  num_rsf_edge_[component] =
249  rfs_tria_v_[component]->NumRefShapeFunctions(1);
250  num_rsf_tria_[component] =
251  rfs_tria_v_[component]->NumRefShapeFunctions(0);
252  }
253  if (rfs_quad_v_[component] != nullptr) {
254  // probe reference finite element for quads.
255  LF_ASSERT_MSG(rfs_quad_v_[component]->RefEl() == lf::base::RefEl::kQuad(),
256  "Wrong type for quad in component " << component);
257  // initialize number of shape functions associated to entitites.
258  num_rsf_node_[component] =
259  rfs_quad_v_[component]->NumRefShapeFunctions(2);
260  num_rsf_edge_[component] =
261  rfs_quad_v_[component]->NumRefShapeFunctions(1);
262  num_rsf_quad_[component] =
263  rfs_quad_v_[component]->NumRefShapeFunctions(0);
264  }
265  if (rfs_edge_v_[component] != nullptr) {
266  LF_ASSERT_MSG(
267  rfs_edge_v_[component]->RefEl() == lf::base::RefEl::kSegment(),
268  "Wrong type for edge in component " << component);
269  num_rsf_node_[component] =
270  rfs_edge_v_[component]->NumRefShapeFunctions(1);
271  num_rsf_edge_[component] =
272  rfs_edge_v_[component]->NumRefShapeFunctions(0);
273  }
274 
275  // Compatibility check for numbers of local shape functions associated with
276  // edges. Those must be the same for all reference shape function
277  // descriptions in a component passed to the finite element space.
278  if ((rfs_tria_v_[component] != nullptr) &&
279  (rfs_quad_v_[component] != nullptr)) {
280  LF_ASSERT_MSG((rfs_tria_v_[component]->NumRefShapeFunctions(2)) ==
281  (rfs_quad_v_[component]->NumRefShapeFunctions(2)),
282  "#RSF missmatch on nodes in component"
283  << component << ": "
284  << rfs_tria_v_[component]->NumRefShapeFunctions(2)
285  << "<->"
286  << rfs_quad_v_[component]->NumRefShapeFunctions(2));
287  LF_ASSERT_MSG((rfs_tria_v_[component]->NumRefShapeFunctions(1)) ==
288  (rfs_quad_v_[component]->NumRefShapeFunctions(1)),
289  "#RSF missmatch on edges in component"
290  << component << ": "
291  << rfs_tria_v_[component]->NumRefShapeFunctions(1)
292  << "<->"
293  << rfs_quad_v_[component]->NumRefShapeFunctions(1));
294  }
295  if ((rfs_tria_v_[component] != nullptr) &&
296  (rfs_edge_v_[component] != nullptr)) {
297  LF_ASSERT_MSG((rfs_tria_v_[component]->NumRefShapeFunctions(2)) ==
298  (rfs_edge_v_[component]->NumRefShapeFunctions(1)),
299  "#RSF missmatch on nodes in component"
300  << component << ": "
301  << rfs_tria_v_[component]->NumRefShapeFunctions(2)
302  << "<->"
303  << rfs_edge_v_[component]->NumRefShapeFunctions(1));
304  LF_ASSERT_MSG((rfs_tria_v_[component]->NumRefShapeFunctions(1)) ==
305  (rfs_edge_v_[component]->NumRefShapeFunctions(0)),
306  "#RSF missmatch on edges in component"
307  << component << ": "
308  << rfs_tria_v_[component]->NumRefShapeFunctions(1)
309  << "<->"
310  << rfs_edge_v_[component]->NumRefShapeFunctions(0));
311  }
312  if ((rfs_quad_v_[component] != nullptr) &&
313  (rfs_edge_v_[component] != nullptr)) {
314  LF_ASSERT_MSG((rfs_quad_v_[component]->NumRefShapeFunctions(2)) ==
315  (rfs_edge_v_[component]->NumRefShapeFunctions(1)),
316  "#RSF missmatch on nodes in component"
317  << component << ": "
318  << rfs_quad_v_[component]->NumRefShapeFunctions(2)
319  << "<->"
320  << rfs_edge_v_[component]->NumRefShapeFunctions(1));
321  LF_ASSERT_MSG((rfs_quad_v_[component]->NumRefShapeFunctions(1)) ==
322  (rfs_edge_v_[component]->NumRefShapeFunctions(0)),
323  "#RSF missmatch on edges in component"
324  << component << ": "
325  << rfs_quad_v_[component]->NumRefShapeFunctions(1)
326  << "<->"
327  << rfs_edge_v_[component]->NumRefShapeFunctions(0));
328  }
329  }
330 
331  // initialization of dof handler
332  // collect the number of interior reference shape functions for all
333  // components
334  std::vector<dof_map_t> rsf_layouts(numComponents_);
335  for (size_type component = 0; component < numComponents_; component++) {
336  rsf_layouts[component] = {
337  {lf::base::RefEl::kPoint(), num_rsf_node_[component]},
338  {lf::base::RefEl::kSegment(), num_rsf_edge_[component]},
339  {lf::base::RefEl::kTria(), num_rsf_tria_[component]},
340  {lf::base::RefEl::kQuad(), num_rsf_quad_[component]}};
341  }
342  dofh_p_ = std::make_unique<ProductUniformFEDofHandler>(mesh_p_, rsf_layouts);
343 }

345 template <typename SCALAR>
346 std::shared_ptr<const lf::uscalfe::ScalarReferenceFiniteElement<SCALAR>>
347 ProductUniformFESpace<SCALAR>::ShapeFunctionLayout(lf::base::RefEl ref_el_type,
348  size_type component) const {
349  // Retrive specification of local shape functions for
350  // a certain component.
351  switch (ref_el_type) {
352  case lf::base::RefEl::kPoint(): {
353  return rfs_point_v_[component];
354  }
355  case lf::base::RefEl::kSegment(): {
356  // Null pointer valid return value:
357  // indicates that a shape function description for edges is missing
358  return rfs_edge_v_[component];
359  }
360  case lf::base::RefEl::kTria(): {
361  // Null pointer valid return value:
362  // indicates that a shape function description for triangles is missing.
363  return rfs_tria_v_[component];
364  }
365  case lf::base::RefEl::kQuad(): {
366  // Null pointer valid return value:
367  // indicates that a shape function description for quadrilaterals is
368  // missing.
369  return rfs_quad_v_[component];
370  }
371  default: {
372  LF_ASSERT_MSG(false, "Illegal entity type");
373  }
374  }
375  return nullptr;
376 }
