18 Eigen::MatrixXd PiecewiseConstElementMatrixProvider::Eval(
19  const lf::mesh::Entity &entity) const {
20  // Get the geometry of the entity
21  const auto geom = entity.Geometry();
22  // Compute the global vertex coordinates
23  Eigen::MatrixXd vertices = geom->Global(entity.RefEl().NodeCoords());
24  // Use the vertex coordinates to compute the local normals on the edges
25  const Eigen::Matrix<double, 2, 3> normals =
26  projects::ipdg_stokes::mesh::computeOutwardNormals(entity);
27  // Construct the basis functions from the curl of the standard hat functions
28  const lf::uscalfe::FeLagrangeO1Tria<double> hat_func;
29  const Eigen::MatrixXd ref_grads =
30  hat_func.GradientsReferenceShapeFunctions(Eigen::VectorXd::Zero(2))
31  .transpose();
32  const Eigen::MatrixXd J_inv_trans =
33  geom->JacobianInverseGramian(Eigen::VectorXd::Zero(2));
34  const Eigen::Matrix<double, 2, 3> grads = J_inv_trans * ref_grads;
35  Eigen::Matrix<double, 2, 3> basis_funct;
36  basis_funct << grads.row(1), -grads.row(0);
37  // Fill the element matrix
38  Eigen::MatrixXd elem_mat = Eigen::MatrixXd::Zero(6, 6);
39  // Upper left block
40  elem_mat.block(0, 0, 3, 3).setZero();
41  // Lower left and upper right blocks
42  const auto edges = entity.SubEntities(1);
43  for (int basis_func_idx = 0; basis_func_idx < 3; ++basis_func_idx) {
44  for (int edge_idx = 0; edge_idx < 3; ++edge_idx) {
45  Eigen::Vector2d t;
46  t << -normals(1, edge_idx), normals(0, edge_idx);
47  elem_mat(edge_idx + 3, basis_func_idx) =
48  basis_funct.col(basis_func_idx).transpose() * t;
49  }
50  }
51  elem_mat.block(0, 3, 3, 3) = elem_mat.block(3, 0, 3, 3).transpose();
52  // Lower right block
53  for (int edge_idx = 0; edge_idx < 3; ++edge_idx) {
54  const auto &edge = *edges[edge_idx];
55  elem_mat(edge_idx + 3, edge_idx + 3) = -1 / sigma_;
56  if (modified_) {
57  elem_mat(edge_idx + 3, edge_idx + 3) /=
58  lf::geometry::Volume(*(edge.Geometry()));
59  }
60  // If the edge is not on the boundary, the entry in the full system matrix
61  // will be the sum of two entries of element matrices but we want it still
62  // to be equal to 1/sigma. For this reason, we divide the local contribution
63  // by 2
64  if (!boundary_(edge)) {
65  elem_mat(edge_idx + 3, edge_idx + 3) /= 2;
66  }
67  }
68  return elem_mat;
69 }
