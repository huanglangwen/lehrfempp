117 DpgElementMatrixProvider<SCALAR>::Eval(const lf::mesh::Entity& cell) {
118  // check for nullptrs
119  LF_ASSERT_MSG(extendedStiffnessMatrixProvider_ != nullptr &&
120  gramianProvider_ != nullptr,
121  "nullptr error for some provider");
122  // check that the method is called on an active cell
123  LF_ASSERT_MSG(gramianProvider_->isActive(cell) &&
124  extendedStiffnessMatrixProvider_->isActive(cell),
125  "Eval method called on inactive cell. " << cell);
126 
127  // evaluate extended stiffness matrix B and local Gramian G
128  ElemMat extendedStiffnessMatrix =
129  extendedStiffnessMatrixProvider_->Eval(cell);
130  ElemMat gramian = gramianProvider_->Eval(cell);
131 
132  // perform some size checks.
133  LF_ASSERT_MSG(gramian.rows() == gramian.cols(),
134  "non quadratic gramian of size ("
135  << gramian.rows() << ", " << gramian.cols() << ") on cell "
136  << cell << "\n");
137  LF_ASSERT_MSG(
138  extendedStiffnessMatrix.rows() == gramian.cols(),
139  "size missmatch between gramian & extended Stiffness matrix on cell "
140  << cell);
141 
142  // evaluate DPG element matrix A = B^T G^-1 B
143  return extendedStiffnessMatrix.transpose() *
144  gramian.ldlt().solve(extendedStiffnessMatrix);
145 }
