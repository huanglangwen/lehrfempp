306 GMshFileV4 ReadGmshFileV4(std::string::const_iterator begin,
307  std::string::const_iterator end,
308  const std::string& version, bool is_binary,
309  int size_t_size, int one,
310  const std::string& filename) {
311  LF_VERIFY_MSG(version == "4.1", "Only version 4.1 is supported so far");
312  LF_VERIFY_MSG(size_t_size == sizeof(std::size_t),
313  "size of size_t must be " << sizeof(std::size_t));
314 
315  GMshFileV4 result;
316  result.version_number = version;
317  result.is_binary = is_binary;
318  result.size_t_size = size_t_size;
319 
320  bool succesful = false;
321  if (!is_binary) {
322  succesful = detail::ParseGmshFileV4Text(begin, end, &result);
323  } else {
324  succesful = detail::ParseGmshFileV4Binary(begin, end, one, &result);
325  }
326 
327  LF_VERIFY_MSG(succesful, "Could not parse file " << filename);
328  // LF_VERIFY_MSG(iter == end, "Could not parse all of file " << filename);
329 
330  // transpose all periodic matrices because they are read in column-first mode
331  // but gmsh writes them in row-first mode
332  for (auto& p : result.periodic_links) {
333  if (p.affine_transform) {
334  p.affine_transform->transposeInPlace();
335  }
336  }
337 
338  return result;
339 }
