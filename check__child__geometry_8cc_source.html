18  LF_ASSERT_MSG(geom.RefEl() == ref_pat.RefEl(),
19  "This refinement pattern is not made for a " << geom.RefEl());
20  double h = 1. / ref_pat.LatticeConst();
21 
22  for (auto codim = 0; codim <= geom.RefEl().Dimension(); ++codim) {
23  auto children = geom.ChildGeometry(ref_pat, codim);
24  auto child_polygons = ref_pat.ChildPolygons(codim);
25

27 
28  for (int i = 0; i < children.size(); ++i) {
29  auto &child = children[i];
30  EXPECT_EQ(child->RefEl().Dimension(), geom.DimLocal() - codim);
31  EXPECT_EQ(child->DimGlobal(), geom.DimGlobal());
32  EXPECT_EQ(child->DimLocal(), geom.DimLocal() - codim);
33 
34  if (child->RefEl() == lf::base::RefEl::kPoint()) {
35  // check that child == geom.Global(...)
36  EXPECT_EQ(child_polygons[i].rows(), geom.DimLocal());
37  EXPECT_EQ(child_polygons[i].cols(), 1);
38  auto child_coord = geom.Global(child_polygons[i].cast<double>() * h);
39 
40  auto zero = Eigen::Matrix<double, 0, 1>::Zero();
41  EXPECT_TRUE(child_coord.isApprox(child->Global(zero)));
42  } else {
43  // Check that the mapping child is the same as geom \cdot
44  // refinementPattern
45  auto qr = qr_provider(child->RefEl());
46  auto a = child->Global(qr.Points());
47 
48  std::unique_ptr<lf::geometry::Geometry> ref_pat_geo = nullptr;
49

53  child_polygons[i].cast<double>() * h);
54  break;

58  break;
59  case lf::base::RefEl::kQuad():

62  break;

79 void checkChildGeometryVolume(const lf::geometry::Geometry &geom,
80  const lf::refinement::RefPat &refPat,
81  const lf::base::sub_idx_t &anchor) {
82  // compute volume by means of overkill quadrature
83  auto computeVolume = [](const lf::geometry::Geometry &geom) {
84  const auto qr = lf::quad::make_QuadRule(geom.RefEl(), 23);
85 
86  const auto &points = qr.Points();
87  const auto &weights = qr.Weights();
88  const auto &integrationElements = geom.IntegrationElement(points);
89 
90  double vol = 0.;
91 
92  for (size_t j = 0; j < points.cols(); ++j) {
93  vol += weights(j) * integrationElements(j);
94  }
95 
96  return vol;
97  };
98 
99  const double volume = computeVolume(geom);
100  double refinedVolume = 0.;
101  auto children = geom.ChildGeometry(
102  lf::refinement::Hybrid2DRefinementPattern(geom.RefEl(), refPat, anchor),
103  0);
104 
105  for (auto &childGeom : children) {
106  refinedVolume += computeVolume(*childGeom);
107  }
108 
109  switch (refPat) {
110  case lf::refinement::RefPat::rp_nil:
111  EXPECT_EQ(refinedVolume, 0.)
112  << refPat << " should not produce any children";
113  break;
114  default:
115  EXPECT_FLOAT_EQ(volume, refinedVolume)
116  << "Parent and children volumes differ for " << refPat;
117  }
118 }
